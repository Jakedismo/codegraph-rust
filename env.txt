# ----------------------------------

# CODEGRAPH_EMBEDDING_MODEL=all-minilm:latest
# CODEGRAPH_OLLAMA_URL=http://localhost:11434

# Jina: Cloud-based embeddings with automatic reranking
# Recommended for production - fast, accurate, and includes code-specific models
# CODEGRAPH_EMBEDDING_PROVIDER=jina
# JINA_API_KEY=your-jina-api-key-here
# CODEGRAPH_EMBEDDING_MODEL=jina-embeddings-v4
# CODEGRAPH_EMBEDDING_DIMENSION=1024
# JINA_API_BASE=https://api.jina.ai/v1
# JINA_ENABLE_RERANKING=true
# JINA_RERANKING_MODEL=jina-reranker-v3
# JINA_RERANKING_TOP_N=10
# JINA_LATE_CHUNKING=true
# JINA_TASK=code.query

# CODEGRAPH_EMBEDDING_DIMENSION=1536
